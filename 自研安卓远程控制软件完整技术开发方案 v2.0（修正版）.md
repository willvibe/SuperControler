## 1. 整体架构设计

### 1.1 为什么放弃直接对接hbbs协议

RustDesk Server（hbbs）使用**protobuf二进制协议**，客户端必须实现完整的`RendezvousMessage`消息集才能与其通信。这等同于重新实现RustDesk客户端的信令层，工作量巨大且维护困难。

**修正方案：** 在VPS上部署一个轻量级**自研Go WebSocket信令服务器**（约350行代码），完全自主协议，配合UDP打洞和WebSocket中继。RustDesk Server可作为备选中继，但信令层完全自主。

### 1.3 视频数据流（关键修正：零拷贝路径）

```
【原方案（错误）】
Display → Bitmap.getPixels() → 手动ARGB→YUV转换 → InputBuffer.put() → MediaCodec编码
  CPU消耗极高，帧率<5fps

【修正方案（正确）】
Display → VirtualDisplay → [Surface BufferQueue] → MediaCodec编码器
                                ↑ 零拷贝，全硬件加速
encoderSurface = MediaCodec.createInputSurface()
VirtualDisplay.setSurface(encoderSurface)  // 显示内容直接流入编码器
```

```go
// server/server.go
package main

import (
    "encoding/json"
    "log"
    "net"
    "net/http"
    "strings"
    "sync"

    "github.com/gorilla/websocket"
)

// ─── 消息类型 ────────────────────────────────────────────────────────────────

type MsgType string

const (
    MsgRegister   MsgType = "register"    // 被控端注册
    MsgConnect    MsgType = "connect"     // 主控端发起连接
    MsgRegistered MsgType = "registered"  // 注册成功响应
    MsgPunchInfo  MsgType = "punch_info"  // NAT打洞信息
    MsgP2PFail    MsgType = "p2p_fail"   // P2P失败，转中继
    MsgRelay      MsgType = "relay"       // 中继数据（base64 payload）
    MsgError      MsgType = "error"
)

type Envelope struct {
    Type    MsgType         `json:"type"`
    From    string          `json:"from,omitempty"`
    To      string          `json:"to,omitempty"`
    Payload json.RawMessage `json:"payload,omitempty"`
}

type RegisterPayload struct {
    DeviceID  string `json:"device_id"`
    PublicKey string `json:"public_key"` // ECDH公钥 base64
    UDPPort   int    `json:"udp_port"`   // 客户端绑定的UDP端口（用于打洞）
}

type ConnectPayload struct {
    TargetID  string `json:"target_id"`
    PublicKey string `json:"public_key"`
    UDPPort   int    `json:"udp_port"`
}

type PunchInfoPayload struct {
    PeerIP     string `json:"peer_ip"`
    PeerPort   int    `json:"peer_port"`
    PeerPubKey string `json:"peer_pubkey"`
    Role       string `json:"role"` // "controlled" or "controller"
    SessionKey string `json:"session_key"` // 用于双方标识同一个会话
}

// ─── 客户端 ──────────────────────────────────────────────────────────────────

type Client struct {
    conn      *websocket.Conn
    deviceID  string
    pubKey    string
    publicIP  string
    udpPort   int
    peerID    string   // 当前配对的对端ID
    send      chan []byte
    mu        sync.Mutex
    closed    bool
}

func (c *Client) Send(env Envelope) {
    data, err := json.Marshal(env)
    if err != nil {
        return
    }
    c.mu.Lock()
    defer c.mu.Unlock()
    if !c.closed {
        select {
        case c.send <- data:
        default:
            // 发送缓冲满，丢弃（视频中继时不应发生，控制消息很小）
        }
    }
}

func (c *Client) writePump() {
    defer c.conn.Close()
    for data := range c.send {
        if err := c.conn.WriteMessage(websocket.TextMessage, data); err != nil {
            break
        }
    }
}

// ─── 服务器 ──────────────────────────────────────────────────────────────────

type Server struct {
    clients  map[string]*Client
    mu       sync.RWMutex
    upgrader websocket.Upgrader
}

func NewServer() *Server {
    return &Server{
        clients: make(map[string]*Client),
        upgrader: websocket.Upgrader{
            ReadBufferSize:  1024 * 64,
            WriteBufferSize: 1024 * 64,
            CheckOrigin: func(r *http.Request) bool { return true },
        },
    }
}

func (s *Server) HandleWS(w http.ResponseWriter, r *http.Request) {
    conn, err := s.upgrader.Upgrade(w, r, nil)
    if err != nil {
        log.Printf("upgrade error: %v", err)
        return
    }

    client := &Client{
        conn:     conn,
        publicIP: extractIP(r.RemoteAddr),
        send:     make(chan []byte, 512),
    }

    go client.writePump()

    defer func() {
        s.removeClient(client)
    }()

    for {
        _, data, err := conn.ReadMessage()
        if err != nil {
            break
        }
        var env Envelope
        if json.Unmarshal(data, &env) != nil {
            continue
        }
        s.dispatch(client, &env)
    }
}

func (s *Server) dispatch(from *Client, env *Envelope) {
    switch env.Type {

    case MsgRegister:
        var p RegisterPayload
        if json.Unmarshal(env.Payload, &p) != nil || p.DeviceID == "" {
            return
        }
        from.deviceID = p.DeviceID
        from.pubKey   = p.PublicKey
        from.udpPort  = p.UDPPort

        s.mu.Lock()
        s.clients[p.DeviceID] = from
        s.mu.Unlock()

        log.Printf("[register] id=%s ip=%s udp=%d", p.DeviceID, from.publicIP, p.UDPPort)

        payload, _ := json.Marshal(map[string]string{"public_ip": from.publicIP})
        from.Send(Envelope{Type: MsgRegistered, Payload: payload})

    case MsgConnect:
        var p ConnectPayload
        if json.Unmarshal(env.Payload, &p) != nil || p.TargetID == "" {
            return
        }
        from.pubKey  = p.PublicKey
        from.udpPort = p.UDPPort

        s.mu.RLock()
        target, ok := s.clients[p.TargetID]
        s.mu.RUnlock()

        if !ok {
            errPayload, _ := json.Marshal(map[string]string{"message": "device_not_found"})
            from.Send(Envelope{Type: MsgError, Payload: errPayload})
            return
        }

        from.peerID   = p.TargetID
        target.peerID = from.deviceID

        // 生成会话key（双方用同一key标识打洞会话）
        sessionKey := newSessionKey()

        // 通知主控端：被控端的公网地址
        toCtrl, _ := json.Marshal(PunchInfoPayload{
            PeerIP:     target.publicIP,
            PeerPort:   target.udpPort,
            PeerPubKey: target.pubKey,
            Role:       "controller",
            SessionKey: sessionKey,
        })
        // 通知被控端：主控端的公网地址
        toDevice, _ := json.Marshal(PunchInfoPayload{
            PeerIP:     from.publicIP,
            PeerPort:   from.udpPort,
            PeerPubKey: from.pubKey,
            Role:       "controlled",
            SessionKey: sessionKey,
        })

        from.Send(Envelope{Type: MsgPunchInfo, Payload: toCtrl})
        target.Send(Envelope{Type: MsgPunchInfo, Payload: toDevice})
        log.Printf("[connect] %s → %s session=%s", from.deviceID, p.TargetID, sessionKey)

    case MsgP2PFail:
        // 双方任一方P2P失败，切换到WebSocket中继模式
        // 此后的MsgRelay消息将被转发
        log.Printf("[relay mode] %s ↔ %s", from.deviceID, from.peerID)

    case MsgRelay:
        // 转发中继数据（二进制用base64编码放在payload里）
        if from.peerID == "" {
            return
        }
        s.mu.RLock()
        peer, ok := s.clients[from.peerID]
        s.mu.RUnlock()
        if ok {
            env.From = from.deviceID
            peer.Send(*env)
        }
    }
}

func (s *Server) removeClient(c *Client) {
    s.mu.Lock()
    if c.deviceID != "" {
        delete(s.clients, c.deviceID)
    }
    s.mu.Unlock()

    c.mu.Lock()
    if !c.closed {
        c.closed = true
        close(c.send)
    }
    c.mu.Unlock()

    log.Printf("[disconnect] id=%s", c.deviceID)
}

func extractIP(addr string) string {
    host, _, err := net.SplitHostPort(addr)
    if err != nil {
        return strings.TrimSpace(addr)
    }
    return host
}

func newSessionKey() string {
    b := make([]byte, 8)
    _, _ = rand.Read(b)
    return fmt.Sprintf("%x", b)
}
```

```kotlin
// core:root/src/main/java/ScreenCaptureManager.kt

/**
 * 使用系统权限 CAPTURE_VIDEO_OUTPUT 创建 VirtualDisplay
 * 需要通过 Root 预先授予此权限（见 App.grantRequiredPermissions）
 *
 * 关键优化：VirtualDisplay 直接连接到 MediaCodec 的输入 Surface
 * 整个路径：Display → VirtualDisplay → [SurfaceTexture/BufferQueue] → MediaCodec
 * 无CPU介入，全程GPU/硬件加速
 */
class ScreenCaptureManager(private val context: Context) {

    private var virtualDisplay: VirtualDisplay? = null
    private val displayManager = context.getSystemService(DisplayManager::class.java)

    /**
     * @param encoderSurface MediaCodec.createInputSurface() 返回的Surface
     */
    fun startCapture(
        width: Int,
        height: Int,
        dpi: Int,
        encoderSurface: Surface
    ): Boolean {
        // 检查权限（已通过Root授予）
        if (context.checkSelfPermission("android.permission.CAPTURE_VIDEO_OUTPUT")
            != PackageManager.PERMISSION_GRANTED) {
            Log.e("ScreenCapture", "CAPTURE_VIDEO_OUTPUT permission not granted")
            return false
        }

        virtualDisplay = displayManager.createVirtualDisplay(
            "RemoteCtrlCapture",
            width,
            height,
            dpi,
            encoderSurface,                          // ← 直接连接编码器Surface
            DisplayManager.VIRTUAL_DISPLAY_FLAG_AUTO_MIRROR  // 镜像主显示
        )

        return virtualDisplay != null
    }

    fun stopCapture() {
        virtualDisplay?.release()
        virtualDisplay = null
    }

    /**
     * 获取设备实际屏幕参数
     */
    fun getDisplayInfo(): Triple<Int, Int, Int> {
        val display = displayManager.getDisplay(Display.DEFAULT_DISPLAY)
        val metrics = DisplayMetrics()
        display.getRealMetrics(metrics)   // getRealMetrics 包含状态栏区域（真实物理分辨率）
        return Triple(metrics.widthPixels, metrics.heightPixels, metrics.densityDpi)
    }
}
```

### 4.3 视频编码（重大修正：Surface输入模式 + SPS/PPS提取）

```kotlin
// core:media/src/main/java/VideoEncoder.kt

class VideoEncoder(
    private val width: Int,
    private val height: Int,
    private val fps: Int = 30,
    private val bitrateBps: Int = 3_000_000
) {
    // 编码后的回调
    var onEncodedFrame: ((ByteArray, Long, Boolean) -> Unit)? = null
    // SPS/PPS配置数据回调（首帧前触发，必须先发给解码端）
    var onCodecConfig: ((sps: ByteArray, pps: ByteArray) -> Unit)? = null
    // 错误回调
    var onError: ((Exception) -> Unit)? = null

    private var codec: MediaCodec? = null
    private var outputJob: Job? = null
    private val scope = CoroutineScope(Dispatchers.IO + SupervisorJob())

    /**
     * 初始化编码器并返回输入Surface
     * ★ 必须在 startCapture 之前调用，拿到Surface后再传给VirtualDisplay
     */
    fun prepare(): Surface {
        val format = MediaFormat.createVideoFormat(MediaFormat.MIMETYPE_VIDEO_AVC, width, height).apply {
            setInteger(MediaFormat.KEY_COLOR_FORMAT,
                MediaCodecInfo.CodecCapabilities.COLOR_FormatSurface) // ← Surface输入模式
            setInteger(MediaFormat.KEY_BIT_RATE, bitrateBps)
            setInteger(MediaFormat.KEY_FRAME_RATE, fps)
            setInteger(MediaFormat.KEY_I_FRAME_INTERVAL, 2)          // 每2秒一个IDR帧
            setInteger(MediaFormat.KEY_BITRATE_MODE,
                MediaCodecInfo.EncoderCapabilities.BITRATE_MODE_VBR)
            // 低延迟配置
            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R) {
                setInteger(MediaFormat.KEY_LOW_LATENCY, 1)
            }
            setInteger("latency", 0)
            setInteger(MediaFormat.KEY_PRIORITY, 0)                  // 实时优先级
        }

        val encoder = MediaCodec.createEncoderByType(MediaFormat.MIMETYPE_VIDEO_AVC)
        encoder.configure(format, null, null, MediaCodec.CONFIGURE_FLAG_ENCODE)

        // ★ 必须在 configure 之后、start 之前调用
        val inputSurface = encoder.createInputSurface()

        codec = encoder
        return inputSurface    // 返回给调用方，传入 VirtualDisplay
    }

    fun start() {
        val c = codec ?: throw IllegalStateException("Call prepare() first")
        c.start()
        outputJob = scope.launch { drainOutputLoop(c) }
    }

    private suspend fun drainOutputLoop(c: MediaCodec) {
        val info = MediaCodec.BufferInfo()
        var sps: ByteArray? = null
        var pps: ByteArray? = null

        while (isActive) {
            val idx = try {
                c.dequeueOutputBuffer(info, 10_000L)
            } catch (e: Exception) {
                onError?.invoke(e)
                break
            }

            when {
                idx == MediaCodec.INFO_OUTPUT_FORMAT_CHANGED -> {
                    // ★ 关键：提取SPS/PPS（编解码器配置数据）
                    val fmt = c.outputFormat
                    sps = fmt.getByteBuffer("csd-0")?.toByteArray()
                    pps = fmt.getByteBuffer("csd-1")?.toByteArray()
                    if (sps != null && pps != null) {
                        onCodecConfig?.invoke(sps, pps)
                        Log.d("VideoEncoder", "SPS(${sps.size}B) + PPS(${pps.size}B) extracted")
                    }
                }

                idx >= 0 -> {
                    val buf = c.getOutputBuffer(idx) ?: run {
                        c.releaseOutputBuffer(idx, false); return@run
                    }
                    val isConfig    = info.flags and MediaCodec.BUFFER_FLAG_CODEC_CONFIG != 0
                    val isKeyFrame  = info.flags and MediaCodec.BUFFER_FLAG_KEY_FRAME != 0
                    val isEos       = info.flags and MediaCodec.BUFFER_FLAG_END_OF_STREAM != 0

                    if (!isConfig && info.size > 0) {
                        val data = ByteArray(info.size)
                        buf.position(info.offset)
                        buf.get(data)
                        onEncodedFrame?.invoke(data, info.presentationTimeUs, isKeyFrame)
                    }

                    c.releaseOutputBuffer(idx, false)
                    if (isEos) break
                }

                idx == MediaCodec.INFO_TRY_AGAIN_LATER -> {
                    delay(1)    // 让出CPU
                }
            }
        }
    }

    /** 动态调整码率（根据网络RTT反馈） */
    fun updateBitrate(newBitrateBps: Int) {
        codec?.setParameters(Bundle().apply {
            putInt(MediaCodec.PARAMETER_KEY_VIDEO_BITRATE, newBitrateBps)
        })
    }

    /** 请求立即输出关键帧（网络丢包后恢复用） */
    fun requestKeyFrame() {
        codec?.setParameters(Bundle().apply {
            putInt(MediaCodec.PARAMETER_KEY_REQUEST_SYNC_FRAME, 0)
        })
    }

    fun stop() {
        outputJob?.cancel()
        scope.cancel()
        try { codec?.stop() } catch (_: Exception) {}
        try { codec?.release() } catch (_: Exception) {}
        codec = null
    }

    private fun ByteBuffer.toByteArray(): ByteArray {
        val pos = position()
        val arr = ByteArray(remaining())
        get(arr)
        position(pos)
        return arr
    }
}
```

```kotlin
class InputInjector(private val context: Context) {

    // ─── 触摸 ──────────────────────────────────────────────────────────────────

     * 使用物理像素绝对坐标（0,0 为屏幕左上角，包含状态栏）

    fun swipe(x1: Int, y1: Int, x2: Int, y2: Int, durationMs: Int = 300) =
        execRoot("input swipe $x1 $y1 $x2 $y2 $durationMs")

    fun longPress(x: Int, y: Int) =
        execRoot("input swipe $x $y $x $y 600")

    fun scroll(x: Int, y: Int, dx: Int, dy: Int) =
        execRoot("input swipe $x $y ${x + dx} ${y + dy} 200")

    // ─── 按键 ──────────────────────────────────────────────────────────────────

    fun key(keyCode: Int) = execRoot("input keyevent $keyCode")
    fun back()    = key(KeyEvent.KEYCODE_BACK)
    fun home()    = key(KeyEvent.KEYCODE_HOME)
    fun recents() = key(KeyEvent.KEYCODE_APP_SWITCH)
    fun power()   = key(KeyEvent.KEYCODE_POWER)

    // ─── 文字输入（修正：支持中文/Emoji/特殊符号）─────────────────────────────

    /**
     * ✅ 正确方案：通过剪贴板 + 粘贴快捷键实现任意文字输入
     * 避免 `input text` 不支持非ASCII字符的问题
     */
    fun inputText(text: String) {
        // 1. Root权限设置剪贴板内容
        Shell.cmd(
            // 使用content命令写入剪贴板（Android 7+）
            """content call --uri content://settings/system \
              --method set_clip_data --extra label:s:rc --extra text:s:${shellEscape(text)}"""
        ).submit { result ->
            if (result.isSuccess) {
                // 2. 发送Ctrl+V粘贴
                Shell.cmd("input keyevent --longpress 279").exec()  // 279 = KEYCODE_PASTE
            } else {
                // 备用：am broadcast方式
                inputTextFallback(text)
            }
        }
    }

    private fun inputTextFallback(text: String) {
        // 使用ADB剪贴板广播（某些ROM支持）
        val escaped = shellEscape(text)
        Shell.cmd(
            "am broadcast -a clipper.set -e text $escaped",
            "input keyevent 279"
        ).exec()
    }

    private fun shellEscape(text: String): String {
        // 用单引号包裹，转义内部单引号
        return "'${text.replace("'", "'\\''")}'"
    }

    // ─── 执行 ──────────────────────────────────────────────────────────────────

    private fun execRoot(cmd: String) {
        Shell.cmd(cmd).submit { result ->
            if (!result.isSuccess) {
                Log.w("InputInjector", "Failed: $cmd → ${result.err}")
            }
        }
    }
}
```

### 4.5 被控端Service（修正版）

```kotlin
// feature:controlled/src/main/java/ControlledService.kt

class ControlledService : Service() {

    companion object {
        const val NOTIFICATION_ID = 2001
        const val CHANNEL_ID = "sync_channel"           // 不要用"remote"等敏感词

        fun start(context: Context) {
            context.startForegroundService(Intent(context, ControlledService::class.java))
        }
        fun stop(context: Context) {
            context.stopService(Intent(context, ControlledService::class.java))
        }
    }

    private val scope = CoroutineScope(Dispatchers.IO + SupervisorJob())
    private lateinit var captureManager: ScreenCaptureManager
    private lateinit var encoder: VideoEncoder
    private lateinit var injector: InputInjector
    private lateinit var signalingClient: SignalingClient

    // 屏幕真实参数
    private var screenWidth = 0
    private var screenHeight = 0
    private var screenDpi = 0

    override fun onCreate() {
        super.onCreate()
        startForeground(NOTIFICATION_ID, buildNotification())    // 必须在5秒内调用

        captureManager = ScreenCaptureManager(this)
        injector = InputInjector(this)

        // 获取屏幕真实分辨率（含物理分辨率，不受密度缩放影响）
        val (w, h, dpi) = captureManager.getDisplayInfo()
        screenWidth = w
        screenHeight = h
        screenDpi = dpi

        // 编码分辨率：长边不超过1280，保持比例（减少带宽）
        val (encW, encH) = calcEncodeSize(w, h, 1280)

        // ★ 正确初始化顺序：prepare() → startCapture() → start()
        encoder = VideoEncoder(encW, encH, fps = 30, bitrateBps = 3_000_000)
        val encoderSurface = encoder.prepare()  // 先拿Surface

        // 屏幕直接渲染到编码器Surface（零拷贝）
        val captureOk = captureManager.startCapture(encW, encH, screenDpi, encoderSurface)
        if (!captureOk) {
            Log.e("ControlledService", "Screen capture failed, stopping service")
            stopSelf()
            return
        }

        encoder.start()

        // 连接信令服务器
        signalingClient = SignalingClient(BuildConfig.SIGNAL_SERVER)
        scope.launch { signalingClient.connect(DeviceIdManager.getDeviceId(this@ControlledService)) }

        setupCallbacks()
    }

    private fun setupCallbacks() {
        // SPS/PPS提取后发给对端（连接建立时发送）
        encoder.onCodecConfig = { sps, pps ->
            signalingClient.sendCodecConfig(sps, pps)
        }

        // 编码帧发给对端
        encoder.onEncodedFrame = { data, pts, isKeyFrame ->
            signalingClient.sendVideoFrame(data, pts, isKeyFrame)
        }

        // 收到控制事件
        signalingClient.onControlEvent = { event ->
            handleControlEvent(event)
        }

        // 收到关键帧请求
        signalingClient.onKeyFrameRequest = {
            encoder.requestKeyFrame()
        }

        // 码率调整（由主控端根据RTT反馈）
        signalingClient.onBitrateAdjust = { bps ->
            encoder.updateBitrate(bps)
        }
    }

    private fun handleControlEvent(event: ControlEventProto) {
        // 使用全屏物理像素坐标系（0,0 = 屏幕左上角）
        val absX = (event.x * screenWidth).toInt()
        val absY = (event.y * screenHeight).toInt()

        when (event.type) {
            EventType.TAP        -> injector.tap(absX, absY)
            EventType.SWIPE      -> injector.swipe(
                absX, absY,
                (event.x2 * screenWidth).toInt(),
                (event.y2 * screenHeight).toInt(),
                event.durationMs
            )
            EventType.LONG_PRESS -> injector.longPress(absX, absY)
            EventType.SCROLL     -> injector.scroll(
                absX, absY,
                (event.dx * screenWidth).toInt(),
                (event.dy * screenHeight).toInt()
            )
            EventType.KEY        -> injector.key(event.keyCode)
            EventType.TEXT       -> injector.inputText(event.text)
            EventType.BACK       -> injector.back()
            EventType.HOME       -> injector.home()
            EventType.RECENTS    -> injector.recents()
            else -> {}
        }
    }

    /** 计算等比例缩放的编码分辨率（长边不超过maxLong） */
    private fun calcEncodeSize(w: Int, h: Int, maxLong: Int): Pair<Int, Int> {
        val ratio = maxLong.toFloat() / maxOf(w, h)
        if (ratio >= 1f) return Pair(w, h)
        // 确保宽高都是16的倍数（H.264编码要求）
        return Pair(
            (w * ratio).toInt().roundTo16(),
            (h * ratio).toInt().roundTo16()
        )
    }

    private fun Int.roundTo16() = (this + 15) / 16 * 16

    private fun buildNotification(): Notification {
        val channel = NotificationChannel(
            CHANNEL_ID, "数据同步", NotificationManager.IMPORTANCE_LOW
        ).apply { setShowBadge(false); setSound(null, null) }
        getSystemService(NotificationManager::class.java).createNotificationChannel(channel)

        return NotificationCompat.Builder(this, CHANNEL_ID)
            .setContentTitle("同步服务")           // 不暴露真实功能
            .setContentText("正在运行")
            .setSmallIcon(android.R.drawable.stat_sys_download)
            .setPriority(NotificationCompat.PRIORITY_MIN)
            .setOngoing(true)
            .build()
    }

    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int) = START_STICKY

    override fun onBind(intent: Intent?) = null

    override fun onDestroy() {
        scope.cancel()
        captureManager.stopCapture()
        encoder.stop()
        signalingClient.disconnect()
        super.onDestroy()
    }
}
```

```kotlin
        val format = MediaFormat.createVideoFormat(MediaFormat.MIMETYPE_VIDEO_AVC, width, height).apply {
            setByteBuffer("csd-0", ByteBuffer.wrap(sps))
            setByteBuffer("csd-1", ByteBuffer.wrap(pps))
        }

        val decoder = MediaCodec.createDecoderByType(MediaFormat.MIMETYPE_VIDEO_AVC)
        decoder.configure(format, surface, null, 0)
        decoder.start()
        codec = decoder
        configured = true
        Log.d("VideoDecoder", "Configured ${width}x${height}")
    }

    /**
     * 解码一帧并直接渲染到Surface（不经过CPU）
     */
    fun decodeFrame(data: ByteArray, pts: Long, isKeyFrame: Boolean) {
        val c = codec ?: return
        if (!configured) return

        val idx = c.dequeueInputBuffer(10_000L)
        if (idx < 0) return

        val buf = c.getInputBuffer(idx) ?: return
        buf.clear()
        buf.put(data)

        val flags = if (isKeyFrame) MediaCodec.BUFFER_FLAG_KEY_FRAME else 0
        c.queueInputBuffer(idx, 0, data.size, pts, flags)

        // 非阻塞轮询输出（不等待，避免积压）
        drainOutput(c)
    }

    private fun drainOutput(c: MediaCodec) {
        while (true) {
            val outIdx = c.dequeueOutputBuffer(info, 0)
            when {
                outIdx >= 0 -> {
                    // true = 渲染到Surface（不经过CPU内存）
                    c.releaseOutputBuffer(outIdx, true)
                }
                outIdx == MediaCodec.INFO_TRY_AGAIN_LATER -> break
                outIdx == MediaCodec.INFO_OUTPUT_FORMAT_CHANGED -> {
                    Log.d("VideoDecoder", "Output format changed: ${c.outputFormat}")
                }
                else -> break
            }
        }
    }

    fun stop() {
        configured = false
        try { codec?.stop() } catch (_: Exception) {}
        try { codec?.release() } catch (_: Exception) {}
        codec = null
    }
}
```

```kotlin
        val relX = ((event.x - offsetX) / scaleX).coerceIn(0f, 1f)
        val relY = ((event.y - offsetY) / scaleY).coerceIn(0f, 1f)

            MotionEvent.ACTION_DOWN -> {
                pointerDownX = relX; pointerDownY = relY
                pointerDownTime = SystemClock.elapsedRealtime()
                isScrolling = false

                val distY = Math.abs(relY - pointerDownY)

                    isScrolling = true

                        relX - pointerDownX, relY - pointerDownY))

                val distX = Math.abs(relX - pointerDownX)
                val distY = Math.abs(relY - pointerDownY)

                        sendControlEvent(newTapEvent(pointerDownX, pointerDownY))
                    }
                } else if (isScrolling) {
                    sendControlEvent(newSwipeEndEvent())
                }
                isScrolling = false
            }
        }
    }
```

```kotlin
// core:network/src/main/java/DataChannel.kt

/**
 * 统一数据通道接口
 * 实现类：UdpDataChannel（P2P直连）、WssRelayChannel（WebSocket中继）
 */
interface DataChannel {
    val isConnected: Boolean
    val rttMs: Long         // 最近一次RTT

    suspend fun send(frame: Frame)
    fun close()

    var onFrame: ((Frame) -> Unit)?
    var onDisconnected: (() -> Unit)?
}

// ─── UDP P2P 通道 ─────────────────────────────────────────────────────────────

class UdpDataChannel(
    private val peerAddress: InetSocketAddress
) : DataChannel {

    override var isConnected = false
    override var rttMs = 0L
    override var onFrame: ((Frame) -> Unit)? = null
    override var onDisconnected: (() -> Unit)? = null

    private val socket = DatagramSocket()
    private val crypto = SessionCrypto()
    private val scope  = CoroutineScope(Dispatchers.IO + SupervisorJob())
    private var lastPingSentMs = 0L
    private var sendSeq = 0L

    fun start(sharedKey: ByteArray) {
        crypto.setSharedKey(sharedKey)
        isConnected = true
        scope.launch { receiveLoop() }
        scope.launch { heartbeatLoop() }
    }

    override suspend fun send(frame: Frame) {
        val seqFrame = frame.toBuilder().setSeq(sendSeq++).setTsMs(System.currentTimeMillis()).build()
        val plaintext = seqFrame.toByteArray()
        val ciphertext = crypto.encrypt(plaintext)

        // UDP最大有效负载约65507字节；视频帧可能超过，需要分片
        if (ciphertext.size <= MAX_UDP_PAYLOAD) {
            val packet = DatagramPacket(ciphertext, ciphertext.size, peerAddress)
            withContext(Dispatchers.IO) { socket.send(packet) }
        } else {
            sendFragmented(ciphertext, seqFrame.seq)
        }
    }

    private suspend fun sendFragmented(data: ByteArray, seq: Long) {
        val totalFrags = (data.size + FRAG_SIZE - 1) / FRAG_SIZE
        for (i in 0 until totalFrags) {
            val start = i * FRAG_SIZE
            val end   = minOf(start + FRAG_SIZE, data.size)
            val frag  = buildFragHeader(seq, i, totalFrags) + data.copyOfRange(start, end)
            val packet = DatagramPacket(frag, frag.size, peerAddress)
            withContext(Dispatchers.IO) { socket.send(packet) }
        }
    }

    private suspend fun receiveLoop() {
        val buf = ByteArray(MAX_UDP_PAYLOAD + FRAG_HEADER_SIZE)
        val packet = DatagramPacket(buf, buf.size)
        while (scope.isActive) {
            try {
                withContext(Dispatchers.IO) { socket.receive(packet) }
                val raw = buf.copyOf(packet.length)

                if (isFragmented(raw)) {
                    reassembleFragment(raw)
                } else {
                    processDecrypted(crypto.decrypt(raw))
                }
            } catch (e: Exception) {
                if (!scope.isActive) break
            }
        }
    }

    private fun processDecrypted(data: ByteArray) {
        val frame = Frame.parseFrom(data)
        if (frame.type == FrameType.HEARTBEAT) {
            // 计算RTT
            rttMs = System.currentTimeMillis() - frame.tsMs
            return
        }
        onFrame?.invoke(frame)
    }

    private suspend fun heartbeatLoop() {
        while (scope.isActive) {
            delay(2000)
            val hb = Frame.newBuilder()
                .setType(FrameType.HEARTBEAT)
                .setTsMs(System.currentTimeMillis())
                .build()
            try { send(hb) } catch (_: Exception) {}
        }
    }

    override fun close() {
        isConnected = false
        scope.cancel()
        socket.close()
    }

    companion object {
        const val MAX_UDP_PAYLOAD = 1400      // 安全MTU
        const val FRAG_SIZE       = 1380
        const val FRAG_HEADER_SIZE = 8        // [seq:4][fragIdx:2][totalFrags:2]
    }
}

// ─── WebSocket 中继通道 ───────────────────────────────────────────────────────

class WssRelayChannel(
    private val wsClient: WebSocket,          // 复用信令WebSocket连接
    private val peerId: String,
    private val crypto: SessionCrypto
) : DataChannel {

    override var isConnected = true
    override var rttMs = 0L
    override var onFrame: ((Frame) -> Unit)? = null
    override var onDisconnected: (() -> Unit)? = null

    private var sendSeq = 0L

    override suspend fun send(frame: Frame) {
        val seqFrame = frame.toBuilder().setSeq(sendSeq++).setTsMs(System.currentTimeMillis()).build()
        val ciphertext = crypto.encrypt(seqFrame.toByteArray())

        // 通过信令服务器中继（base64编码）
        val envelope = JSONObject().apply {
            put("type", "relay")
            put("to", peerId)
            put("payload", Base64.encodeToString(ciphertext, Base64.NO_WRAP))
        }
        wsClient.send(envelope.toString())
    }

    fun onRelayData(base64Data: String) {
        val ciphertext = Base64.decode(base64Data, Base64.NO_WRAP)
        val frame = Frame.parseFrom(crypto.decrypt(ciphertext))
        if (frame.type == FrameType.HEARTBEAT) {
            rttMs = System.currentTimeMillis() - frame.tsMs
        } else {
            onFrame?.invoke(frame)
        }
    }

    override fun close() { isConnected = false }
}
```

### 6.3 信令客户端 + 连接状态机（修正版）

```kotlin
// core:network/src/main/java/SignalingClient.kt

class SignalingClient(private val serverUrl: String) {

    // 对外回调
    var onHandshake: ((HandshakeInfo) -> Unit)? = null
    var onVideoFrame: ((ByteArray, Long, Boolean) -> Unit)? = null
    var onControlEvent: ((ControlEvent) -> Unit)? = null
    var onKeyFrameRequest: (() -> Unit)? = null
    var onBitrateAdjust: ((Int) -> Unit)? = null
    var onRttUpdate: ((Long) -> Unit)? = null

    private var ws: WebSocket? = null
    private var dataChannel: DataChannel? = null
    private val crypto = SessionCrypto()
    private var myId = ""
    private var peerId = ""
    private var myPublicIp = ""
    private val scope = CoroutineScope(Dispatchers.IO + SupervisorJob())
    private var reconnectAttempts = 0

    // ─── 状态机 ──────────────────────────────────────────────────────────────

    sealed class State {
        object Idle : State()
        object Connecting : State()           // 正在连接信令服务器
        object Registered : State()           // 已注册，等待对端
        object Punching : State()             // 正在UDP打洞
        object Connected : State()            // P2P连通
        object Relaying : State()             // WebSocket中继模式
        data class Error(val msg: String) : State()
    }

    private var state: State = State.Idle
    private val stateLock = Mutex()

    // ─── 被控端入口 ──────────────────────────────────────────────────────────

    suspend fun connect(deviceId: String) {
        myId = deviceId
        connectWebSocket()
    }

    // ─── 主控端入口 ──────────────────────────────────────────────────────────

    suspend fun connectToDevice(myId: String, targetId: String, pin: String) {
        this.myId   = myId
        this.peerId = targetId
        connectWebSocket()
        // WebSocket连接成功后在onOpen中发送 connect 消息
    }

    // ─── WebSocket连接 ───────────────────────────────────────────────────────

    private suspend fun connectWebSocket() {
        setState(State.Connecting)

        val client = OkHttpClient.Builder()
            .pingInterval(20, TimeUnit.SECONDS)
            .build()

        val request = Request.Builder()
            .url(serverUrl)
            .build()

        val listener = object : WebSocketListener() {
            override fun onOpen(ws: WebSocket, response: Response) {
                this@SignalingClient.ws = ws
                onWsOpen(ws)
            }
            override fun onMessage(ws: WebSocket, text: String) = onWsMessage(text)
            override fun onFailure(ws: WebSocket, t: Throwable, response: Response?) {
                Log.e("SignalingClient", "WS failure: ${t.message}")
                scheduleReconnect()
            }
            override fun onClosed(ws: WebSocket, code: Int, reason: String) {
                scheduleReconnect()
            }
        }

        client.newWebSocket(request, listener)
    }

    private fun onWsOpen(ws: WebSocket) {
        reconnectAttempts = 0

        // 绑定UDP Socket用于打洞
        val udpSocket = DatagramSocket()
        val udpPort = udpSocket.localPort

        // 发送注册/连接消息
        val payload = JSONObject().apply {
            if (peerId.isEmpty()) {
                // 被控端：注册
                put("device_id", myId)
                put("public_key", crypto.getPublicKeyBase64())
                put("udp_port", udpPort)
            } else {
                // 主控端：连接到目标
                put("target_id", peerId)
                put("public_key", crypto.getPublicKeyBase64())
                put("udp_port", udpPort)
            }
        }
        val type = if (peerId.isEmpty()) "register" else "connect"
        ws.send(JSONObject().apply { put("type", type); put("payload", payload) }.toString())

        setState(State.Registered)
    }

    private fun onWsMessage(text: String) {
        val msg = JSONObject(text)
        when (msg.getString("type")) {

            "registered" -> {
                myPublicIp = msg.getJSONObject("payload").getString("public_ip")
                Log.d("Signaling", "Registered, public IP: $myPublicIp")
            }

            "punch_info" -> {
                val payload  = msg.getJSONObject("payload")
                val peerIp   = payload.getString("peer_ip")
                val peerPort = payload.getInt("peer_port")
                val peerKey  = payload.getString("peer_pubkey")
                val sessionKey = payload.getString("session_key")

                // 计算ECDH共享密钥
                crypto.computeSharedKey(Base64.decode(peerKey, Base64.NO_WRAP))

                // 启动P2P打洞
                scope.launch {
                    val p2pSuccess = attemptUdpPunch(peerIp, peerPort, sessionKey)
                    if (!p2pSuccess) {
                        Log.i("Signaling", "P2P failed, switching to WebSocket relay")
                        // 通知服务器切换中继
                        ws?.send(JSONObject().apply { put("type", "p2p_fail") }.toString())
                        // 启动WebSocket中继通道
                        val relayChannel = WssRelayChannel(ws!!, peerId, crypto)
                        dataChannel = relayChannel
                        setState(State.Relaying)
                        // 如果是被控端，发送握手信息
                        sendHandshakeIfControlled()
                    }
                }
            }

            "relay" -> {
                // 收到中继数据
                val b64Data = msg.getString("payload")
                (dataChannel as? WssRelayChannel)?.onRelayData(b64Data)
            }

            "error" -> {
                val errMsg = msg.getJSONObject("payload").getString("message")
                Log.e("Signaling", "Server error: $errMsg")
                setState(State.Error(errMsg))
            }
        }
    }

    // ─── UDP打洞（修正：双方同步发包）────────────────────────────────────────

    private suspend fun attemptUdpPunch(
        peerIp: String,
        peerPort: Int,
        sessionKey: String
    ): Boolean = withContext(Dispatchers.IO) {
        setState(State.Punching)
        val udpSocket = DatagramSocket()
        val peerAddr = InetSocketAddress(peerIp, peerPort)
        val punchData = "RCPUNCH:$sessionKey".toByteArray()

        // 发送多个打洞包（穿透不同类型NAT）
        repeat(5) {
            val packet = DatagramPacket(punchData, punchData.size, peerAddr)
            try { udpSocket.send(packet) } catch (_: Exception) {}
            delay(100)
        }

        // 等待对端的打洞包（超时4秒）
        udpSocket.soTimeout = 4000
        val recvBuf = DatagramPacket(ByteArray(256), 256)
        return@withContext try {
            udpSocket.receive(recvBuf)
            val recvData = String(recvBuf.data, 0, recvBuf.length)

            if (recvData.startsWith("RCPUNCH:$sessionKey")) {
                Log.i("Signaling", "P2P hole punch SUCCESS! Peer: $peerIp:$peerPort")

                // 建立UDP数据通道
                val udpChannel = UdpDataChannel(peerAddr)
                udpChannel.start(crypto.getSharedKey())
                udpChannel.onFrame = { frame -> processFrame(frame) }
                dataChannel = udpChannel
                setState(State.Connected)

                sendHandshakeIfControlled()
                true
            } else false
        } catch (e: SocketTimeoutException) {
            udpSocket.close()
            false
        }
    }

    // ─── 帧处理 ──────────────────────────────────────────────────────────────

    private fun processFrame(frame: Frame) {
        onRttUpdate?.invoke(dataChannel?.rttMs ?: 0)

        when (frame.type) {
            FrameType.VIDEO_FRAME -> {
                val vf = VideoFrame.parseFrom(frame.payload)
                onVideoFrame?.invoke(vf.data.toByteArray(), vf.pts, vf.isKeyFrame)
            }
            FrameType.CODEC_CONFIG -> {
                val cfg = CodecConfig.parseFrom(frame.payload)
                // 将SPS/PPS包装进HandshakeInfo回调
                val hi = HandshakeInfo.newBuilder()
                    .setSps(cfg.sps).setPps(cfg.pps)
                    .setEncodeWidth(cfg.width).setEncodeHeight(cfg.height)
                    .build()
                onHandshake?.invoke(hi)
            }
            FrameType.CONTROL_EVENT -> {
                val ce = ControlEvent.parseFrom(frame.payload)
                onControlEvent?.invoke(ce)
            }
            FrameType.KEY_FRAME_REQ -> onKeyFrameRequest?.invoke()
            FrameType.BITRATE_ADJUST -> {
                val ba = BitrateAdjust.parseFrom(frame.payload)
                onBitrateAdjust?.invoke(ba.bitrateBps)
            }
            else -> {}
        }
    }

    // ─── 发送方法 ────────────────────────────────────────────────────────────

    fun sendVideoFrame(data: ByteArray, pts: Long, isKeyFrame: Boolean) {
        val vf = VideoFrame.newBuilder()
            .setPts(pts).setIsKeyFrame(isKeyFrame).setData(ByteString.copyFrom(data))
            .build()
        sendFrame(FrameType.VIDEO_FRAME, vf.toByteArray())
    }

    fun sendCodecConfig(sps: ByteArray, pps: ByteArray) {
        val cfg = CodecConfig.newBuilder()
            .setSps(ByteString.copyFrom(sps)).setPps(ByteString.copyFrom(pps))
            .build()
        sendFrame(FrameType.CODEC_CONFIG, cfg.toByteArray())
    }

    fun sendControlEvent(event: ControlEvent) {
        sendFrame(FrameType.CONTROL_EVENT, event.toByteArray())
    }

    fun sendBitrateAdjust(bps: Int) {
        val ba = BitrateAdjust.newBuilder().setBitrateBps(bps).build()
        sendFrame(FrameType.BITRATE_ADJUST, ba.toByteArray())
    }

    private fun sendFrame(type: FrameType, payload: ByteArray) {
        scope.launch {
            val frame = Frame.newBuilder().setType(type).setPayload(ByteString.copyFrom(payload)).build()
            try { dataChannel?.send(frame) } catch (e: Exception) {
                Log.e("SignalingClient", "Send error: ${e.message}")
            }
        }
    }

    // ─── 重连 ────────────────────────────────────────────────────────────────

    private fun scheduleReconnect() {
        val delayMs = minOf(1000L * (1 shl reconnectAttempts), 30_000L)   // 指数退避，最长30秒
        reconnectAttempts++
        Log.i("Signaling", "Reconnect in ${delayMs}ms (attempt $reconnectAttempts)")
        scope.launch {
            delay(delayMs)
            connectWebSocket()
        }
    }

    private fun sendHandshakeIfControlled() {
        // 被控端主动发送握手（包含屏幕信息）
        // 由 ControlledService 调用 sendCodecConfig 触发
    }

    private fun setState(newState: State) {
        state = newState
        Log.d("Signaling", "State → $newState")
    }

    fun disconnect() {
        scope.cancel()
        ws?.close(1000, "disconnect")
        dataChannel?.close()
    }
```

### 7.1 ECDH密钥交换（修正：正确的BouncyCastle API）

```kotlin
// core:crypto/src/main/java/SessionCrypto.kt

import org.bouncycastle.crypto.agreement.ECDHBasicAgreement
import org.bouncycastle.crypto.generators.ECKeyPairGenerator
import org.bouncycastle.crypto.params.*
import org.bouncycastle.crypto.digests.SHA256Digest
import org.bouncycastle.crypto.generators.HKDFBytesGenerator
import org.bouncycastle.crypto.params.HKDFParameters
import org.bouncycastle.jce.ECNamedCurveTable
import org.bouncycastle.jce.provider.BouncyCastleProvider
import java.security.Security

class SessionCrypto {

    companion object {
        init {
            // 注册BouncyCastle Provider（在Application中初始化一次即可）
            if (Security.getProvider("BC") == null) {
                Security.addProvider(BouncyCastleProvider())
            }
        }
        const val HKDF_INFO = "remotectrl-session-v1"
    }

    private val keyPair: AsymmetricCipherKeyPair
    private var sharedKey: ByteArray? = null    // 32字节AES-256密钥

    init {
        // 使用 secp256r1 (NIST P-256) 曲线
        val ecParams = ECNamedCurveTable.getParameterSpec("secp256r1")
        val domainParams = ECDomainParameters(ecParams.curve, ecParams.g, ecParams.n, ecParams.h)

        val keyParams = ECKeyGenerationParameters(domainParams, java.security.SecureRandom())
        val generator = ECKeyPairGenerator()
        generator.init(keyParams)
        keyPair = generator.generateKeyPair()
    }

    /** 获取本端公钥（发给对端） */
    fun getPublicKeyBase64(): String {
        val pubKey = (keyPair.public as ECPublicKeyParameters)
        val encoded = pubKey.q.getEncoded(false)  // 非压缩格式 04|x|y
        return Base64.encodeToString(encoded, Base64.NO_WRAP)
    }

    /** 根据对端公钥计算共享密钥 */
    fun computeSharedKey(peerPubKeyBytes: ByteArray) {
        val ecParams = ECNamedCurveTable.getParameterSpec("secp256r1")
        val domainParams = ECDomainParameters(ecParams.curve, ecParams.g, ecParams.n, ecParams.h)

        val peerPoint = ecParams.curve.decodePoint(peerPubKeyBytes)
        val peerKey   = ECPublicKeyParameters(peerPoint, domainParams)

        val agreement = ECDHBasicAgreement()
        agreement.init(keyPair.private)
        val rawShared = agreement.calculateAgreement(peerKey).toByteArray()

        // ✅ 正确的BouncyCastle HKDF调用
        val hkdf = HKDFBytesGenerator(SHA256Digest())
        hkdf.init(HKDFParameters(
            rawShared,                        // IKM（输入密钥材料）
            null,                             // Salt（不使用）
            HKDF_INFO.toByteArray()           // Info（上下文标识）
        ))
        sharedKey = ByteArray(32).also { hkdf.generateBytes(it, 0, 32) }
        Log.d("SessionCrypto", "Shared key established")
    }

    fun getSharedKey(): ByteArray = sharedKey ?: throw IllegalStateException("Key not established")
    fun setSharedKey(key: ByteArray) { sharedKey = key }

    /** AES-256-GCM 加密 */
    fun encrypt(plaintext: ByteArray): ByteArray {
        val key = sharedKey ?: throw IllegalStateException("Key not established")
        val iv  = ByteArray(12).also { java.security.SecureRandom().nextBytes(it) }

        val cipher = javax.crypto.Cipher.getInstance("AES/GCM/NoPadding")
        cipher.init(
            javax.crypto.Cipher.ENCRYPT_MODE,
            javax.crypto.spec.SecretKeySpec(key, "AES"),
            javax.crypto.spec.GCMParameterSpec(128, iv)
        )
        val ciphertext = cipher.doFinal(plaintext)

        // 格式: [12字节 IV] + [N字节 密文+16字节认证Tag]
        return iv + ciphertext
    }

    /** AES-256-GCM 解密 */
    fun decrypt(data: ByteArray): ByteArray {
        val key        = sharedKey ?: throw IllegalStateException("Key not established")
        val iv         = data.copyOfRange(0, 12)
        val ciphertext = data.copyOfRange(12, data.size)

        val cipher = javax.crypto.Cipher.getInstance("AES/GCM/NoPadding")
        cipher.init(
            javax.crypto.Cipher.DECRYPT_MODE,
            javax.crypto.spec.SecretKeySpec(key, "AES"),
            javax.crypto.spec.GCMParameterSpec(128, iv)
        )
        return cipher.doFinal(ciphertext)
    }
}
```

### 7.2 设备配对PIN机制（新增）

```kotlin
// core:crypto/src/main/java/PairingManager.kt

/**
 * 连接鉴权流程：
 * 1. 被控端显示6位PIN码（首次连接时设置，持久化）
 * 2. 主控端输入PIN
 * 3. PIN通过ECDH共享密钥加密传输，服务器无法解密
 * 4. 被控端验证PIN，成功后允许控制
 */
object PairingManager {

    private const val PREFS_NAME = "pairing"
    private const val KEY_PIN    = "pin_hash"

    /** 初始化PIN（首次运行时生成，或用户手动设置） */
    fun initPin(context: Context, pin: String = generatePin()): String {
        val hash = hashPin(pin)
        context.getSharedPreferences(PREFS_NAME, Context.MODE_PRIVATE)
            .edit().putString(KEY_PIN, hash).apply()
        return pin
    }

    /** 验证PIN */
    fun verifyPin(context: Context, inputPin: String): Boolean {
        val storedHash = context.getSharedPreferences(PREFS_NAME, Context.MODE_PRIVATE)
            .getString(KEY_PIN, null) ?: return false
        return hashPin(inputPin) == storedHash
    }

    private fun generatePin(): String = (100000..999999).random().toString()

    private fun hashPin(pin: String): String {
        val digest = java.security.MessageDigest.getInstance("SHA-256")
        return digest.digest(("remotectrl:$pin").toByteArray())
            .joinToString("") { "%02x".format(it) }
    }
}
```

```kotlin
object DeviceIdManager {

    /**
     * 生成稳定的设备ID
     * 使用多个硬件标识符组合，Root后可获取更稳定的信息
     */
    fun getDeviceId(context: Context): String {
        val prefs = context.getSharedPreferences("device", Context.MODE_PRIVATE)
        var id = prefs.getString("id", null)
        if (id != null) return id

        // 首次生成
        val androidId = Settings.Secure.getString(
            context.contentResolver, Settings.Secure.ANDROID_ID
        ) ?: "unknown"

        // 结合其他硬件信息（Root可获取更多）
        val raw = buildString {
            append(androidId)
            append(Build.BOARD)
            append(Build.BOOTLOADER)
            append(Build.HARDWARE)
        }

        id = sha256(raw).take(12).uppercase()
        prefs.edit().putString("id", id).apply()
        return id
    }

    private fun sha256(input: String): String {
        val digest = java.security.MessageDigest.getInstance("SHA-256")
        return digest.digest(input.toByteArray()).joinToString("") { "%02x".format(it) }
    }
}
```

## 10. 性能调优指南

### 10.1 延迟来源分析

```
总端到端延迟 = 截图延迟 + 编码延迟 + 网络延迟 + 解码延迟 + 渲染延迟

截图延迟:   VirtualDisplay → Surface  ≈ 0ms  （硬件直通，几乎为零）
编码延迟:   Surface → H.264 NAL      ≈ 16-33ms （1-2帧延迟）
网络延迟:   P2P直连                   ≈ 20-80ms
            WebSocket中继             ≈ 50-150ms
解码延迟:   H.264 → Surface          ≈ 16-33ms
渲染延迟:   Surface → 显示           ≈ 16ms    （1帧）

理想P2P总延迟:   ≈ 70-160ms
中继模式总延迟:  ≈ 130-260ms
```

### 10.3 自适应码率控制器

```kotlin
class BitrateController {

    private var currentBps  = 2_000_000
    private val minBps      = 400_000
    private val maxBps      = 8_000_000
    private val rttWindow   = ArrayDeque<Long>(8)

    fun onRtt(rttMs: Long): Int {
        rttWindow.addLast(rttMs)
        if (rttWindow.size > 8) rttWindow.removeFirst()
        val avgRtt = rttWindow.average()

        currentBps = when {
            avgRtt < 40   -> (currentBps * 1.15).toInt().coerceAtMost(maxBps)   // 网络很好，提速
            avgRtt < 100  -> currentBps                                           // 保持
            avgRtt < 200  -> (currentBps * 0.85).toInt().coerceAtLeast(minBps)   // 轻微拥塞
            avgRtt < 400  -> (currentBps * 0.70).toInt().coerceAtLeast(minBps)   // 明显拥塞
            else          -> minBps                                               // 严重拥塞
        }

        return currentBps
    }
}
```

```kotlin
// 2. Service START_STICKY + onTaskRemoved 广播
class ControlledService : Service() {
    override fun onStartCommand(...) = START_STICKY

    override fun onTaskRemoved(rootIntent: Intent) {
        // 任务栏划掉时，发送广播让BootReceiver重启Service
        val restart = Intent("com.yourapp.RESTART_SERVICE")
        sendBroadcast(restart)
    }
}

// 3. 双进程守护（独立push进程）
// AndroidManifest.xml 中声明独立进程
<service android:name=".WatchdogService"
         android:process=":watchdog" />

// 4. MIUI专有：申请自启权限
val intent = Intent().apply {
    component = ComponentName(
        "com.miui.securitycenter",
        "com.miui.permcenter.autostart.AutoStartManagementActivity"
    )
}
startActivity(intent)
```

### Q5: 触控坐标偏移

```
★ 坐标系设计规范（v2.0统一约定）:

被控端发送的屏幕尺寸:   使用 getRealMetrics() 获取的物理分辨率（含状态栏）
坐标原点:               左上角 (0,0)，不减去状态栏高度
相对坐标范围:           [0.0, 1.0] × [0.0, 1.0]

被控端转换（控制事件）:
  absX = relX × physicalWidth    (physicalWidth 来自 getRealMetrics)
  absY = relY × physicalHeight   (physicalHeight 来自 getRealMetrics)
  ✅ input tap absX absY         (input命令接受物理像素坐标，含状态栏)

主控端转换（触摸采集）:
  1. SurfaceView保持与远端相同宽高比（letterbox填充）
  2. 先计算有效显示区域 (displayW, displayH, offsetX, offsetY)
  3. relX = (touchX - offsetX) / displayW
     relY = (touchY - offsetY) / displayH
```

### Q6: Android 14前台Service崩溃

```xml
<service
    android:name=".ControlledService"
    android:foregroundServiceType="mediaProjection|specialUse">
    <!-- Android 14要求specialUse必须加justification -->
    <property
        android:name="android.app.PROPERTY_SPECIAL_USE_FGS_SUBTYPE"
        android:value="screen_sharing" />

// Android 14+ 必须在startForeground中指定type
if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.UPSIDE_DOWN_CAKE) {
    startForeground(
        NOTIFICATION_ID,
        buildNotification(),
        ServiceInfo.FOREGROUND_SERVICE_TYPE_MEDIA_PROJECTION or
        ServiceInfo.FOREGROUND_SERVICE_TYPE_SPECIAL_USE
    )

    startForeground(NOTIFICATION_ID, buildNotification())
}
```
